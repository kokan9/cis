SHOW DATABASES;

CREATE DATABASE PRAC1;

USE PRAC1;

CREATE TABLE usercredentials(
    uid INT(10) UNSIGNED NOT NULL AUTO_INCREMENT, 
    username VARCHAR(255) DEFAULT NULL,
    userpassword VARCHAR(255) DEFAULT NULL,
    PRIMARY KEY (uid),
    UNIQUE KEY username(username)
);
